# ============================================================
# REQUIREMENTS.TXT — список Python-пакетов для бэкенда
#
# Как работает: при развертывании сервера команда
#   pip install -r requirements.txt
# автоматически скачает и установит все пакеты из этого файла.
# Версии зафиксированы (==) чтобы ничего не сломалось
# при выходе новых версий пакетов.
# ============================================================

# FastAPI — фреймворк для создания веб-сервера.
# Это «скелет» сайта: он принимает HTTP-запросы от браузера
# (когда кто-то заходит на сайт) и возвращает HTML-страницы,
# JSON-данные или файлы фото.
# Аналог в других языках: Express.js (Node.js), Flask (Python).
fastapi==0.115.0

# Uvicorn — HTTP-сервер, который запускает FastAPI.
# FastAPI — это только код, он не может сам слушать порт 8000.
# Uvicorn — это «официант», который стоит у двери (порта),
# принимает заказы (запросы) и передаёт их на кухню (FastAPI).
# [standard] — дополнительные фишки: uvloop (быстрый цикл событий),
# watchfiles (автоперезагрузка при изменении кода).
uvicorn[standard]==0.30.0

# Pillow — библиотека для обработки изображений.
# Используется для: уменьшения фото (миниатюры),
# конвертации в WebP/AVIF (современные форматы, меньше размер).
# Без неё: сайт отдаёт оригинальные фото (3-5 МБ каждое),
# что медленно для посетителей.
Pillow==11.0.0

# pillow-avif-plugin — плагин для Pillow, добавляет поддержку
# формата AVIF. AVIF — самый компактный формат фото (на 50%
# меньше WebP при том же качестве). Поддерживается в Chrome,
# Firefox, Safari 16+. Не все браузеры понимают AVIF,
# поэтому есть fallback (WebP → JPEG).
pillow-avif-plugin>=1.4.0

# Jinja2 — шаблонизатор. Превращает HTML-шаблоны с переменными
# в готовые HTML-страницы. Например, шаблон:
#   <h1>{{ title }}</h1>
# с данными title="Портфолио" превратится в:
#   <h1>Портфолио</h1>
# Без Jinja2 пришлось бы вручную склеивать строки HTML в Python.
# >=3.1.0 — минимальная версия, можно новее (совместимы).
jinja2>=3.1.0
